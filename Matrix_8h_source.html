253 template <typename T, size_t N>
254 T det(SquareMatrix<T, N> M)
255 {
256  switch (N) {
257  case 0:
258  throw exceptions::Exception("zero-size matric", "determinant");
259  case 1:
260  return M[0][0];
261  case 2:
262  return M[0][0] * M[1][1] - M[0][1] * M[1][0];
263  case 3:
264  return M[0][0] * M[1][1] * M[2][2]
265  + M[0][1] * M[1][2] * M[2][0]
266  + M[0][2] * M[1][0] * M[2][1]
267  - M[0][2] * M[1][1] * M[2][0]
268  - M[0][0] * M[1][2] * M[2][1]
269  - M[0][1] * M[1][0] * M[2][2];
270  default:
271  T d(0);
272  for (size_t i = 0; i < N; ++i)
273  d += cofactor(M, i, 0);
274  return d;
275  }
276 }
