126 template <typename T>
127 CoordinateSystem<T, 3> helicityFrame(const ThreeVector<T>& V, const CoordinateSystem<T, 3>& C)
128 {
129  constexpr T epsilon = T(5) * std::numeric_limits<T>::epsilon();
130 
131  // if V is 0, return C
132  if (norm(V) <= epsilon)
133  return C;
134 
135  CoordinateSystem<double, 3> vC;
136 
137  // Set Z to V direction
138  vC[2] = unit(V);
139 
140  // calc angle between new Z and old Z
141  auto theta = angle(vC[2], C[2]);
142 
143  // if Z direction is same, return C
144  if (std::abs(theta) <= epsilon)
145  return C;
146 
147  // if Z direction is opposite, return C rotated 180 degrees around Y axis
148  if (std::abs(theta - pi<double>()) <= epsilon)
149  return rotation<T>(C[1], rad(180.)) * C;
150 
151  // Y := (C's Z) cross Z
152  vC[1] = unit(cross(C[2], vC[2]));
153 
154  // X := Y cross Z (right-handed)
155  vC[0] = cross(vC[1], vC[2]);
156 
157  return vC;
158 }
